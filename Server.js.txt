const express = require('express');
const axios = require('axios');
const LRU = require('lru-cache');
const cors = require('cors');

const app = express();
app.use(cors());
app.use(express.json());

// Cache: max 500 items, 5-minute TTL
const cache = new LRU({ max: 500, ttl: 1000 * 60 * 5 });

// TheMealDB API base (test key 1)
const THEMEAL_API_BASE = 'https://www.themealdb.com/api/json/v1/1';

// Helper: fetch with cache
async function fetchWithCache(key, url) {
  if (cache.has(key)) return cache.get(key);
  const resp = await axios.get(url);
  cache.set(key, resp.data);
  return resp.data;
}

// REST endpoints

// Search meals by name
app.get('/api/search', async (req, res) => {
  try {
    const q = req.query.q || '';
    const key = `search:${q}`;
    const url = `${THEMEAL_API_BASE}/search.php?s=${encodeURIComponent(q)}`;
    const data = await fetchWithCache(key, url);
    res.json(data);
  } catch (err) {
    console.error(err.message);
    res.status(500).json({ error: 'Failed to fetch search results' });
  }
});

// List categories
app.get('/api/categories', async (req, res) => {
  try {
    const key = 'categories:list';

// Start server
const PORT = process.env.PORT || 4000;
app.listen(PORT, () => console.log(`Backend listening on http://localhost:${PORT}`));
